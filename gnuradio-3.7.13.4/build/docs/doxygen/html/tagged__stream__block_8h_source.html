51  /*!
52  * \brief Parse all tags on the first sample of a PDU, return the
53  * number of items per input and prune the length tags.
54  *
55  * In most cases, you don't need to override this, unless the
56  * number of items read is not directly coded in one single tag.
57  *
58  * Default behaviour:
59  * - Go through all input ports
60  * - On every input port, search for the tag with the key specified in \p length_tag_key
61  * - Copy that value as an int to the corresponding position in \p n_input_items_reqd
62  * - Remove the length tag.
63  *
64  * \param[in] tags All the tags found on the first item of every input port.
65  * \param[out] n_input_items_reqd Number of items which will be read from every input
66  */
67  virtual void parse_length_tags(const std::vector<std::vector<tag_t> > &tags,
68  gr_vector_int &n_input_items_reqd);

114  /*!
115  * \brief Just like gr::block::general_work, but makes sure the input is valid
116  *
117  * The user must override work to define the signal processing
118  * code. Check the documentation for general_work() to see what
119  * happens here.
120  *
121  * Like gr::sync_block, this calls consume() for you (it consumes
122  * ninput_items[i] items from the i-th port).
123  *
124  * A note on tag propagation: The PDU length tags are handled by
125  * other functions, but all other tags are handled just as in any
126  * other \p gr::block. So, most likely, you either set the tag
127  * propagation policy to TPP_DONT and handle the tag propagation
128  * manually, or you propagate tags through the scheduler and don't
129  * do anything here.
130  *
131  * \param noutput_items The size of the writable output buffer
132  * \param ninput_items The exact size of the items on every input for this particular PDU.
133  * These will be consumed if a length tag key is provided!
134  * \param input_items See gr::block
135  * \param output_items See gr::block
136  */
137  virtual int work(int noutput_items,
138  gr_vector_int &ninput_items,
139  gr_vector_const_void_star &input_items,
140  gr_vector_void_star &output_items) = 0;
